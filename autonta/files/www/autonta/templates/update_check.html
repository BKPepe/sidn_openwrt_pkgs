$def with (lk, dst, update_check_failed, update_available, current_version, currently_beta, other_version, update_version, update_info) $var lk = lk
$lk.SIDNLABS $lk.VALIBOX $lk.UPDATE
$if update_check_failed:

$lk.UPDATE_CHECK_FAILED

$lk.UPDATE_CHECK_CURRENT_VERSION(current_version)

$lk.UPDATE_TRY_AGAIN

$else: $if update_available:

$lk.UPDATE_CHECK_AVAILABLE((update_version, current_version)) 

                $:update_info
$lk.UPDATE_KEEP_SETTINGS $if currently_beta:

$lk.UPDATE_INSTALL_CURRENT

$lk.UPDATE_INSTALL_LATEST(other_version)

$else:

$lk.UPDATE_INSTALL_CURRENT

$lk.UPDATE_INSTALL_LATEST_BETA(other_version)

$lk.UPDATE_INSTALL_START

$else:

$lk.UPDATE_CHECK_NO_UPDATE(current_version)

$lk.UPDATE_TRY_AGAIN

$lk.UPDATE_INSTALL_AGAIN

$lk.UPDATE_KEEP_SETTINGS $if currently_beta:

$lk.UPDATE_INSTALL_REINSTALL

$lk.UPDATE_INSTALL_LATEST(other_version)

$else:

$lk.UPDATE_INSTALL_REINSTALL

$lk.UPDATE_INSTALL_LATEST_BETA(other_version)

$lk.UPDATE_INSTALL_START
