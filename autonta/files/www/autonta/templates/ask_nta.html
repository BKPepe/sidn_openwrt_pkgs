$def with (lk, dst, name, names, err, nta_disabled) $var lk = lk

$lk.DNSSEC_ERROR_NAME(name)

$lk.DNSSEC_ERROR_NAME_LONG(err.target_dname)

$lk.DNSSEC_ERROR_REASON

$lk.DNSSEC_ERROR_SERVER(err.auth_server)

$lk.DNSSEC_ERROR_ERROR(err.err_msg)

$lk.DNSSEC_ERROR_RECORD((err.fail_type, err.fail_dname))

$if not nta_disabled:

$lk.DNSSEC_ERROR_SET_NTA_DESCRIPTION

$lk.DNSSEC_ERROR_NOT_IF_NOT_TRUSTED

$for name in names:

$lk.DNSSEC_ERROR_SET_NTA(name)
